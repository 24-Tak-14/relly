import random


class BuzznD:
    def __init__(self):
        self.media_outlets = {
            "newspapers": ["The Daily Blitz", "Gridiron Gazette", "Sunday Scaries"],
            "podcasts": ["No Mercy Sports", "Tough Talk Football", "The Bleacher Report"]
        }
        self.question_templates = {
            "intrusive": [

                "Would you say your failure today is due to lack of talent or lack of effort?",
                "When will you finally admit you're not good enough for this league?"
            ],
            # Other question categories...
        }
    
    def generate_press_questions(self, team: Dict, game_result: Dict) -> List[str]:
        """Generate provocative press conference questions"""
        questions = []
        num_questions = 3
        
        # Select question types based on game outcome
        if game_result['won']:
            question_style = "skeptical"
        else:
            question_style = "humiliating"
        
        for _ in range(num_questions):
            template = random.choice(self.question_templates[question_style])
            question = template.format(
                player=team['key_player'],
                score=game_result['score']
            )
            questions.append(question)
        
        return questions
    
    def generate_article(self, team: Dict, event: Dict) -> Dict:

        return {
            "headline": headline,
            "body": self._generate_article_body(team, event),
            "outlet": random.choice(self.media_outlets['newspapers'])
        }
