                logging.info(f"Successfully logged in to {self.site}.")
                return True
            else:
                logging.error(f"Login to {self.site} failed. Check credentials.")
                return False
        except requests.exceptions.RequestException as e:
            logging.error(f"Login request failed for {self.site}: {e}")
            return False

    def optimize_videos(self, video_id, title, description, tags, category):
        """Optimizes a video on the site."""
        if not self.login():
            return False
        optimize_urls = {

            "xvideos": f"https://www.xvideos.com/edit_video/{video_id}",
            "xhamster": f"https://xhamster.com/edit_video/{video_id}"

        optimize_data = {
            "title": title,
            "description": description,
            "tags": tags,
            "category": category,
            "submit": "Save Changes"
        }
        try:
            response = self.session.post(optimize_urls[self.site], data=optimize_data)
            response.raise_for_status()
            if response.status_code == 200:
                logging.info(f"Successfully optimized video {video_id} on {self.site}.")
                return True
            else:
                logging.error(f"Failed to optimize video {video_id} on {self.site}.")
                return False
        except requests.exceptions.RequestException as e:
            logging.error(f"Optimization request failed for {self.site}: {e}")
            return False

            logging.error(f"Request exception: {e}")
    
        driver.quit()
        if __name__ == "__main__":
           main()
